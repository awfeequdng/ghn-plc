\newcommand{\sizearrowleft}[3]
{
  \coordinate (x1) at ($#1+(-1.5,0)$);
  \coordinate (x2) at ($#1+(-1.5,0)$);
  \coordinate (y1) at ($#2+(-1.5,0)$);
  \coordinate (y2) at ($#2+(-1.5,0)$);
  \draw[-]#1 -- (x1);
  \draw[-]#2 -- (y1);
  \draw[latex'-latex'] (x2) -- node[left, rotate=90, anchor=south]{\Large #3} (y2);
}

\newcommand{\sizearrowbelow}[3]
{
  \coordinate (x1) at ($#1+(0,-1.5)$);
  \coordinate (x2) at ($#1+(0,-1.5)$);
  \coordinate (y1) at ($#2+(0,-1.5)$);
  \coordinate (y2) at ($#2+(0,-1.5)$);
  \draw[-]#1 -- (x1);
  \draw[-]#2 -- (y1);
  \draw[latex'-latex'] (x2) -- node[above]{\Large #3} (y2);
}
